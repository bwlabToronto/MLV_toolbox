0021 function sphere_points = sample_sphere_2D(number_of_samples)
0022 
0023 sphere_points = zeros(number_of_samples,2);
0024 
0025 alpha = (2*pi)/(number_of_samples);
0026 
0027 
0028 
0029 
0030 for i = 1 : number_of_samples
0031     
0032     sphere_points(i,1) = cos(alpha*(i-1));
0033     sphere_points(i,2) = sin(alpha*(i-1));
0034     
0035 end
