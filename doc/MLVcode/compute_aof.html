0026 %initialize Q
0027 [m,n] = size(D);
0028 nOfSamples = size(sphere_points,1);

0030 %for each point on the sphere create the normal vector
0031 normals = zeros(size(sphere_points));
0032 fluxImage = zeros(m,n);

0034 for t = 1 : nOfSamples
0035     normals(t,:) = sphere_points(t,:);
0036 end
0037 
0038 %scale sphere_points by its radius (epsilon)
0039 sphere_points = sphere_points * epsilon;
0040 
0041 for i = 1 : m
0042     for j = 1 : n       
0043             flux_value = 0;
0044             if(D(i,j) > -1.5)            
0045                 if( i > epsilon && j > epsilon && i < m - epsilon && j < n - epsilon )
0046                     %sum over dot product of normal and the gradient vector field (q-dot)
0047                     for ind = 1 : nOfSamples
0048                                                 
0049                         % a point on the sphere
0050                         px = i+sphere_points(ind,1)+0.5;
0051                         py = j+sphere_points(ind,2)+0.5;
0052                         
0053                         
0054                         
0055                         
0056                         %the indices of the grid cell that sphere points
0057                         %fall into
0058                         cI = floor(i+sphere_points(ind,1)+0.5);
0059                         cJ = floor(j+sphere_points(ind,2)+0.5);
0060                                                
0061                         
0062                         %closest point on the boundary to that sphere point
0063                         [bx,by] = ind2sub(size(D),IDX(cI,cJ));
0064 
0065                         % the vector connect them
0066                         qq = [bx,by] - [px,py];
0067                         d = norm(qq);
0068                         if(d~=0)
0069                             qq = qq / d;

0072                         end
0073                         flux_value = flux_value + dot(qq,normals(ind,:));
0074                     end
0075                 end

0077             end
